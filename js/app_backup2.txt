// Video control functions
function initVideoControls() {
    const video = document.getElementById('hero-video');
    const videoContainer = document.querySelector('.video-container');
    const videoLoading = document.querySelector('.video-loading');
    
    if (video) {
        // Video loaded successfully
        video.addEventListener('loadeddata', function() {
            console.log('âœ… Video loaded successfully');
            videoLoading.classList.add('video-loaded');
            
            // Auto-adjust video size based on aspect ratio
            adjustVideoSize(video);
        });
        
        // Video error handling
        video.addEventListener('error', function(e) {
            console.log('âŒ Video failed to load:', e);
            videoLoading.classList.add('video-error');
            videoLoading.classList.add('video-loaded');
        });
        
        // Video playing
        video.addEventListener('playing', function() {
            console.log('ðŸŽ¥ Video is playing');
        });
    }
}

function adjustVideoSize(video) {
    // Get video dimensions
    const videoAspectRatio = video.videoWidth / video.videoHeight;
    const containerAspectRatio = window.innerWidth / 280; // Hero section height
    
    // Adjust zoom based on aspect ratio
    if (videoAspectRatio > containerAspectRatio) {
        // Video is wider than container
        video.parentElement.classList.add('video-zoom-small');
    } else {
        // Video is taller than container
        video.parentElement.classList.add('video-zoom-medium');
    }
}

// Add this to your existing DOMContentLoaded function
document.addEventListener('DOMContentLoaded', function() {
    // ... your existing code ...
    
    // Initialize video controls
    initVideoControls();
    
    // Re-adjust video on resize
    window.addEventListener('resize', function() {
        const video = document.getElementById('hero-video');
        if (video && video.readyState >= 2) { // HAVE_CURRENT_DATA or better
            adjustVideoSize(video);
        }
    });
});
